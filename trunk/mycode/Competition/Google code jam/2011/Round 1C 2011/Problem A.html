1 #include <fstream>
 2  #include <iostream>
 3  using namespace std;
 4  ifstream fin("dataA.in");
 5  ofstream fout("dataA.out");
 6  const int maxn = 100 + 1;
 7  int p[maxn] = {0};
 8  char r[maxn]={0};
 9  int t;
10  int n;
11  int ans;
12  
13  
14  void work()
15  {
16      int i,b,o,bn,on;
17      b = o = bn = on = 1;
18      for(i = 1; ; i ++)
19      {
20          while(r[bn] == 'O' && bn <= n)bn ++;
21          while(r[on] == 'B' && on <= n)on ++;
22          if(bn <= n)
23          {
24              if(p[bn] > b)
25                  b ++;
26              else if(p[bn] < b)
27                  b --;
28              else if(bn < on)
29                  bn ++;
30          }
31          if(on <= n)
32          {
33              if(p[on] > o)
34                  o ++;
35              else if(p[on] < o)
36                  o --;
37              else if(on < bn)
38                  on ++;
39          }
40          //cout << b << ' ' << o << endl;
41          if(bn > n && on > n)
42              break;
43      }
44      ans = i;
45  }
46  
47  void init()
48  {
49      fin >> n;
50      for(int i = 1; i <= n; i ++)
51          fin >> r[i] >> p[i];
52  }
53  
54  void output()
55  {
56      fout << ans << endl;
57  }
58  
59  int main()
60  {
61      int i,j;
62      fin >> t;
63      for(i=1;i<=t;i++)
64      {
65          init();
